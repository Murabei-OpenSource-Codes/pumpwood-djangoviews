 114    def list(self, request):
 115        """
 116        View function to list objects with pagination.
 117
 118        Number of objects are limited by
 119        settings.REST_FRAMEWORK['PAGINATE_BY']. To get next page, use
 120        exclude_dict['pk__in': [list of the received pks]] to get more
 121        objects.
 122
 123        Use to limit the query .query.filter_by_dict function.
 124
 125        :param request.data['filter_dict']: Dictionary passed as
 126                                            objects.filter(**filter_dict)
 127        :type request.data['filter_dict']: dict
 128        :param request.data['exclude_dict']: Dictionary passed as
 129                                             objects.exclude(**exclude_dict)
 130        :type request.data['exclude_dict']: dict
 131        :param request.data['order_by']: List passed as
 132                                             objects.order_by(*order_by)
 133        :type request.data['order_by']: list
 134        :return: A list of objects using list_serializer
 135        """
 136        try:
 137            request_data = request.data
 138            limit = request_data.pop("limit", None)
 139            list_paginate_limit = limit or self.list_paginate_limit
 140
 141            # Serializer parameters
 142            fields = request_data.pop("fields", None)
 143            default_fields = request_data.pop("default_fields", False)
 144            foreign_key_fields = request_data.pop("foreign_key_fields", False)
 145
 146            arg_dict = {'query_set': self.service_model.objects.all()}
 147            arg_dict.update(request_data)
 148            query_set = filter_by_dict(**arg_dict)[:list_paginate_limit]
 149
 150            return Response(self.serializer(
 151                query_set, many=True, fields=fields,
 152                foreign_key_fields=foreign_key_fields,
 153                default_fields=default_fields).data)
 154
 155        except TypeError as e:
 156            raise exceptions.PumpWoodQueryException(message=str(e))

 230    def retrieve_file(self, request, pk: int):
 231        """
 232        Read file without stream.
 233
 234        Args:
 235            pk (int): Pk of the object to save file field.
 236            file_field(str): File field to receive stream file.
 237
 238        Returns:
 239            A stream of bytes with da file.
 240        """
 241        if self.storage_object is None:
 242            raise exceptions.PumpWoodForbidden(
 243                "storage_object not set")
 244
 245        file_field = request.query_params.get('file-field', None)
 246        if file_field not in self.file_fields.keys():
 247            msg = (
 248                "'{file_field}' must be set on file_fields "
 249                "dictionary.").format(file_field=file_field)
 250            raise exceptions.PumpWoodForbidden(msg)
 251        obj = self.service_model.objects.get(id=pk)
 252
 253        file_path = getattr(obj, file_field)
 254        if isinstance(file_path, FieldFile):
 255            file_path = file_path.name
 256
 257        if not file_path:
 258            raise exceptions.PumpWoodObjectDoesNotExist(
 259                "field [{}] is not set at object".format(file_field))
 260        file_data = self.storage_object.read_file(file_path)
 261        file_name = os.path.basename(file_path)
 262
 263        response = HttpResponse(content=BytesIO(file_data["data"]))
 264        response['Content-Type'] = file_data["content_type"]
 265        response['Content-Disposition'] = \
 266            'attachment; filename=%s' % file_name
 267        return response

 304    def remove_file_field(self, request, pk: int) -> bool:
 305        """
 306        Remove file field.
 307
 308        Args:
 309            pk (int): pk of the object.
 310        Kwargs:
 311            No kwargs for this function.
 312        Raises:

 315            PumpWoodException: Propagates exceptions from storage_objects.

 319            raise exceptions.PumpWoodForbidden(

 324        if file is None:
 325            raise exceptions.PumpWoodObjectDoesNotExist(

 327        else:
 328            file_path = file.name
 329        setattr(obj, file_field, None)
 330        obj.save()
 331
 332        try:
 333            self.storage_object.delete_file(file_path)
 334            return Response(True)
 335        except Exception as e:
 336            raise exceptions.PumpWoodException(str(e))

 338    def save(self, request):
 339        """
 340        Save and update object acording to request.data.
 341
 342        Object will be updated if request.data['pk'] is not None.
 343
 344        :param dict request.data: Object representation as
 345            self.retrieve_serializer
 346        :raise PumpWoodException: 'Object model class diferent from
 347            {service_model} : {service_model}' request.data['service_model']
 348                not the same as self.service_model.__name__
 349        """
 350        request_data: dict = None
 351        if "application/json" in request.content_type.lower():
 352            request_data = request.data
 353        else:
 354            request_data = request.data.dict()
 355            json_data = json.loads(request_data.pop("__json__", '{}'))
 356            request_data.update(json_data)
 357
 358        data_pk = request_data.get('pk')
 359        saved_obj = None
 360        for field in self.file_fields.keys():
 361            request_data.pop(field, None)
 362
 363        # update
 364        if data_pk:
 365            data_to_update = self.service_model.objects.get(pk=data_pk)
 366            serializer = self.serializer(
 367                data_to_update, data=request_data,
 368                context={'request': request})
 369            saved_obj = save_serializer_instance(serializer)
 370            response_status = status.HTTP_200_OK
 371        # save
 372        else:
 373            serializer = self.serializer(
 374                data=request_data, context={'request': request})
 375            saved_obj = save_serializer_instance(serializer)
 376            response_status = status.HTTP_201_CREATED
 377
 378        # Uploading files
 379        object_errors = {}
 380        file_save_time = datetime.datetime.utcnow().strftime(
 381            "%Y-%m-%dT%Hh%Mm%Ss")
 382        for field in self.file_fields.keys():
 383            field_errors = []
 384            if field in request.FILES:
 385                file = request.FILES[field]
 386
 387                file_name = secure_filename(file.name)
 388                field_errors.extend(self._allowed_extension(
 389                    filename=file_name,
 390                    allowed_extensions=self.file_fields[field]))
 391
 392                filename = "{}___{}___{}".format(
 393                    str(saved_obj.id).zfill(15),
 394                    file_save_time,
 395                    file_name)
 396                if len(field_errors) != 0:
 397                    object_errors[field] = field_errors
 398                else:
 399                    model_class = self.service_model.__name__.lower()
 400                    file_path = '{model_class}__{field}/'.format(
 401                        model_class=model_class, field=field)
 402                    storage_filepath = self.storage_object.write_file(
 403                        file_path=file_path, file_name=filename,
 404                        data=file.read(),
 405                        content_type=file.content_type,
 406                        if_exists='overide')
 407                    setattr(saved_obj, field, storage_filepath)
 408
 409        if object_errors != {}:
 410            message = "error when saving object: " \
 411                if data_pk is None else "error when updating object: "
 412            payload = object_errors
 413            message_to_append = []
 414            for key, value in object_errors.items():
 415                message_to_append.append(key + ", " + str(value))
 416            message = message + "; ".join(message_to_append)
 417            raise exceptions.PumpWoodObjectSavingException(
 418                message=message, payload=payload)
 419        saved_obj.save()
 420
 421        if self.microservice is not None and self.trigger:
 422            # Process ETLTrigger for the model class
 423            self.microservice.login()
 424            if data_pk is None:
 425                self.microservice.execute_action(
 426                    "ETLTrigger", action="process_triggers", parameters={
 427                        "model_class": self.service_model.__name__.lower(),
 428                        "type": "create",
 429                        "pk": None,
 430                        "action_name": None})
 431            else:
 432                self.microservice.execute_action(
 433                    "ETLTrigger", action="process_triggers", parameters={
 434                        "model_class": self.service_model.__name__.lower(),
 435                        "type": "update",
 436                        "pk": saved_obj.pk,
 437                        "action_name": None})
 438
 439        # Overhead, serializando e deserializando o objecto
 440        return Response(
 441            self.serializer(saved_obj).data, status=response_status)

 491    def execute_action(self, request, action_name, pk=None):
 492        """Execute action over object or class using parameters."""
 493        parameters = request.data
 494        actions = self.get_actions()
 495        rest_action_names = list(actions.keys())
 496
 497        if action_name not in rest_action_names:
 498            message = (
 499                "There is no method {action} in rest actions "
 500                "for {class_name}").format(
 501                    action=action_name,
 502                    class_name=self.service_model.__name__)
 503            raise exceptions.PumpWoodForbidden(
 504                message=message, payload={"action_name": action_name})
 505
 506        action = getattr(self.service_model, action_name)
 507        if pk is None and not action.action_object.is_static_function:
 508            msg_template = (
 509                "Action [{action}] at model [{class_name}] is not "
 510                "a classmethod and not pk provided.")
 511            message = msg_template.format(
 512                action=action_name,
 513                class_name=self.service_model.__name__)
 514            raise exceptions.PumpWoodActionArgsException(
 515                message=message, payload={"action_name": action_name})
 516
 517        if pk is not None and action.action_object.is_static_function:
 518            msg_template = (
 519                "Action [{action}] at model [{class_name}] is a"
 520                "classmethod and pk provided.")
 521            message = msg_template.format(
 522                action=action_name,
 523                class_name=self.service_model.__name__)
 524            raise exceptions.PumpWoodActionArgsException(
 525                message=message, payload={"action_name": action_name})
 526
 527        object_dict = None
 528        action = None
 529        if pk is not None:
 530            model_object = self.service_model.objects.filter(pk=pk).first()
 531            if model_object is None:
 532                message_template = (
 533                    "Requested object {service_model}[{pk}] not found.")
 534                temp_service_model = \
 535                    self.service_model.__name__
 536                message = message_template.format(
 537                    service_model=temp_service_model, pk=pk)
 538                raise exceptions.PumpWoodObjectDoesNotExist(
 539                    message=message, payload={
 540                        "service_model": temp_service_model, "pk": pk})
 541
 542            action = getattr(model_object, action_name)
 543            object_dict = self.serializer(model_object, many=False).data
 544        else:
 545            action = getattr(self.service_model, action_name)
 546
 547        loaded_parameters = load_action_parameters(action, parameters, request)
 548        result = action(**loaded_parameters)
 549
 550        if self.microservice is not None and self.trigger:
 551            self.microservice.login()
 552            self.microservice.execute_action(
 553                "ETLTrigger", action="process_triggers", parameters={
 554                    "model_class": self.service_model.__name__.lower(),
 555                    "type": "action", "pk": pk, "action_name": action_name})
 556
 557        return Response({
 558            'result': result, 'action': action_name,
 559            'parameters': parameters, 'object': object_dict})

 563        """Return field options using serializer."""
 564        fields = cls.service_model._meta.get_fields()
 565        dict_fields = {}
 566        for f in fields:
 567            is_relation = getattr(f, "is_relation", False)
 568            primary_key = getattr(f, "primary_key", False)
 569            if not is_relation or primary_key:
 570                dict_fields[f.column] = f
 571
 572        model_class = cls.service_model.__name__
 573        translation_tag_template = "{model_class}__fields__{field}"
 574
 575        # Get read-only fields from serializer
 576        read_only_fields = getattr(cls.serializer.Meta, "read_only_fields", [])
 577
 578        # Get field serializers
 579        serializer_fields = cls.serializer(
 580            foreign_key_fields=True, related_fields=True).fields
 581
 582        # Get serializers associated with FKs and related models
 583        microservice_fk_dict = {}
 584        microservice_related_dict = {}
 585
 586        all_info = {}
 587        for field_name, field_serializer in serializer_fields.items():
 588            ################################################################
 589            # Do not create relations between models in search description #
 590            is_microservice_fk = isinstance(
 591                field_serializer, MicroserviceForeignKeyField)
 592            is_local_fk = isinstance(
 593                field_serializer, LocalForeignKeyField)
 594            if is_microservice_fk or is_local_fk:
 595                field_key = field_serializer.get_fields_options_key()
 596                microservice_fk_dict[field_key] = field_serializer
 597
 598            is_microservice_related = isinstance(
 599                field_serializer, MicroserviceRelatedField)
 600            is_local_related = isinstance(
 601                field_serializer, LocalRelatedField)
 602            if is_microservice_related or is_local_related:
 603                field_key = field_serializer.get_fields_options_key()
 604                microservice_related_dict[field_key] = field_serializer
 605
 606            f = dict_fields.get(field_serializer.source)
 607            if f is None:
 608                continue
 609            ################################################################
 610
 611            column_info = {}
 612
 613            # Getting correspondent simple type
 614            column_type = f.get_internal_type()
 615            python_type = django_map.get(column_type)
 616            if python_type is None:
 617                msg = (
 618                    "Type [{column_type}] not implemented in map dictionary "
 619                    "django type -> python type")
 620                raise NotImplementedError(
 621                    msg.format(column_type=column_type))
 622
 623            # Getting default value
 624            default = None
 625            f_default = getattr(f, 'default', None)
 626            if f_default != NOT_PROVIDED and f_default is not None:
 627                if callable(f_default):
 628                    default = f_default()
 629                else:
 630                    default = f_default
 631

377            response_status = status.HTTP_201_CREATED

379        # Uploading files
